CREATE TABLE CANDIDATE_CV (
id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
CV_CREATE DATETIME DEFAULT CURRENT_TIMESTAMP,
CV_FILENAME VARCHAR(50),
NAME VARCHAR(50),
SURNAME VARCHAR(30),
EMAIL VARCHAR(100),
PHONE VARCHAR(20),
MOBILE VARCHAR(50),
LANGUAGE VARCHAR(50),
LINKEDIN VARCHAR(50),
SOCIAL VARCHAR(10),
CONFIDENCE INT(3)


)
